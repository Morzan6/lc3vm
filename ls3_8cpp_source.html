199void LC3State::load_image(const std::string &filename) {
200 std::ifstream file(filename, std::ios::binary);
201 if (!file.is_open()) {
202 throw std::runtime_error("Failed to open image file: " + filename);
203 }
204
205 std::uint16_t origin;
206 file.read(reinterpret_cast<char*>(&origin), sizeof(origin));
207 if (file.gcount() != sizeof(origin)) {
208 throw std::runtime_error("Failed to read origin from image file: " + filename);
209 }
210 origin = swap16(origin);
211
212 std::uint16_t max_words_to_read = MEMORY_MAX - origin;
213 std::vector<std::uint16_t> image_data(max_words_to_read);
214
215 file.read(reinterpret_cast<char*>(image_data.data()), max_words_to_read * sizeof(std::uint16_t));
216 size_t words_actually_read = file.gcount() / sizeof(std::uint16_t);
217
218 for (size_t i = 0; i < words_actually_read; ++i) {
219 if (origin + i < MEMORY_MAX) {
220 this->memory.write(origin + i, swap16(image_data[i]));
221 } else {
222 throw std::runtime_error("Image too large for memory at specified origin.");
223 }
224 }
225}

242void LC3State::step() {
243 if (!this->running) return;
244
245 std::uint16_t current_pc = this->reg[R_PC];
246 this->reg[R_PC]++;
247
248 std::uint16_t instruction = this->memory.read(current_pc);
249 std::uint16_t opcode_val = instruction >> 12;
250
251 if (opcode_val >= op_table.size() || !op_table[opcode_val]) {
252 throw std::runtime_error("Illegal or unsupported opcode: " + std::to_string(opcode_val) +
253 " at PC: " + std::to_string(current_pc));
254 }
255
256 op_table[opcode_val](*this, instruction);
257}
