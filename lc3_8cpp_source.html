36template <unsigned op>
37void LC3State::ins(LC3State& state, std::uint16_t instr) {
38 if (op == OP_BR) {
39 std::uint16_t cond_flag_from_instr = (instr >> 9) & 0x7;
40 if (cond_flag_from_instr & state.reg[R_COND]) {
41 state.reg[R_PC] += sign_extend(instr & 0x1FF, 9);
42 }
43 }
44 else if (op == OP_ADD) {
45 std::uint16_t r0 = (instr >> 9) & 0x7;
46 std::uint16_t r1 = (instr >> 6) & 0x7;
47 std::uint16_t imm_flag = (instr >> 5) & 0x1;
48 if (imm_flag) {
49 std::uint16_t imm5 = sign_extend(instr & 0x1F, 5);
50 state.reg[r0] = state.reg[r1] + imm5;
51 } else {
52 std::uint16_t r2 = instr & 0x7;
53 state.reg[r0] = state.reg[r1] + state.reg[r2];
54 }
55 state.update_flags(r0);
56 }
57 else if (op == OP_LD) {
58 std::uint16_t r0 = (instr >> 9) & 0x7;
59 std::uint16_t pc_offset9 = sign_extend(instr & 0x1FF, 9);
60 state.reg[r0] = state.memory.read(state.reg[R_PC] + pc_offset9);
61 state.update_flags(r0);
62 }
63 else if (op == OP_ST) {
64 std::uint16_t r0 = (instr >> 9) & 0x7;
65 std::uint16_t pc_offset9 = sign_extend(instr & 0x1FF, 9);
66 state.memory.write(state.reg[R_PC] + pc_offset9, state.reg[r0]);
67 }
68 else if (op == OP_JSR) {
69 state.reg[R_R7] = state.reg[R_PC];
70 std::uint16_t long_flag = (instr >> 11) & 1;
71 if (long_flag) {
72 std::uint16_t long_pc_offset = sign_extend(instr & 0x7FF, 11);
73 state.reg[R_PC] += long_pc_offset;
74 } else {
75 std::uint16_t r1 = (instr >> 6) & 0x7;
76 state.reg[R_PC] = state.reg[r1];
77 }
78 }
79 else if (op == OP_AND) {
80 std::uint16_t r0 = (instr >> 9) & 0x7;
81 std::uint16_t r1 = (instr >> 6) & 0x7;
82 std::uint16_t imm_flag = (instr >> 5) & 0x1;
83 if (imm_flag) {
84 std::uint16_t imm5 = sign_extend(instr & 0x1F, 5);
85 state.reg[r0] = state.reg[r1] & imm5;
86 } else {
87 std::uint16_t r2 = instr & 0x7;
88 state.reg[r0] = state.reg[r1] & state.reg[r2];
89 }
90 state.update_flags(r0);
91 }
92 else if (op == OP_LDR) {
93 std::uint16_t r0 = (instr >> 9) & 0x7;
94 std::uint16_t r1 = (instr >> 6) & 0x7;
95 std::uint16_t offset6 = sign_extend(instr & 0x3F, 6);
96 state.reg[r0] = state.memory.read(state.reg[r1] + offset6);
97 state.update_flags(r0);
98 }
99 else if (op == OP_STR) {
100 std::uint16_t r0 = (instr >> 9) & 0x7;
101 std::uint16_t r1 = (instr >> 6) & 0x7;
102 std::uint16_t offset6 = sign_extend(instr & 0x3F, 6);
103 state.memory.write(state.reg[r1] + offset6, state.reg[r0]);
104 }
105 else if (op == OP_NOT) {
106 std::uint16_t r0 = (instr >> 9) & 0x7;
107 std::uint16_t r1 = (instr >> 6) & 0x7;
108 state.reg[r0] = ~state.reg[r1];
109 state.update_flags(r0);
110 }
111 else if (op == OP_LDI) {
112 std::uint16_t r0 = (instr >> 9) & 0x7;
113 std::uint16_t pc_offset9 = sign_extend(instr & 0x1FF, 9);
114 std::uint16_t effective_address_location = state.reg[R_PC] + pc_offset9;
115 state.reg[r0] = state.memory.read(state.memory.read(effective_address_location));
116 state.update_flags(r0);
117 }
118 else if (op == OP_STI) {
119 std::uint16_t r0 = (instr >> 9) & 0x7;
120 std::uint16_t pc_offset9 = sign_extend(instr & 0x1FF, 9);
121 std::uint16_t effective_address_location = state.reg[R_PC] + pc_offset9;
122 state.memory.write(state.memory.read(effective_address_location), state.reg[r0]);
123 }
124 else if (op == OP_JMP) {
125 std::uint16_t r1 = (instr >> 6) & 0x7;
126 state.reg[R_PC] = state.reg[r1];
127 }
128 else if (op == OP_LEA) {
129 std::uint16_t r0 = (instr >> 9) & 0x7;
130 std::uint16_t pc_offset9 = sign_extend(instr & 0x1FF, 9);
131 state.reg[r0] = state.reg[R_PC] + pc_offset9;
132 state.update_flags(r0);
133 }
134 else if (op == OP_TRAP) {
135 state.reg[R_R7] = state.reg[R_PC];
136 switch (instr & 0xFF) {
137 case TRAP_GETC:
138 {
139 if (state.memory.test_mode) {
140 state.reg[R_R0] = state.memory.read(Keyboard::MR_KBDR);
141 } else {
142 char c_in = 0;
143 if (read(STDIN_FILENO, &c_in, 1) == 1) {
144 state.reg[R_R0] = static_cast<std::uint16_t>(c_in);
145 }
146 }
147 }
148 state.update_flags(R_R0);
149 break;
150 case TRAP_OUT:
151 if (!state.memory.test_mode) {
152 std::cout.put(static_cast<char>(state.reg[R_R0]));
153 std::cout.flush();
154 }
155 break;
156 case TRAP_PUTS: {
157 if (!state.memory.test_mode) {
158 std::uint16_t current_char_addr = state.reg[R_R0];
159 std::uint16_t val = state.memory.read(current_char_addr);
160 while (val != 0) {
161 std::cout.put(static_cast<char>(val));
162 current_char_addr++;
163 val = state.memory.read(current_char_addr);
164 }
165 std::cout.flush();
166 }
167 break;
168 }
169 case TRAP_IN: {
170 if (state.memory.test_mode) {
171 state.reg[R_R0] = state.memory.read(Keyboard::MR_KBDR);
172 } else {
173 std::cout << "Enter a character: ";
174 std::cout.flush();
175 char c_in_trap = 0;
176 if (read(STDIN_FILENO, &c_in_trap, 1) == 1) {
177 std::cout.put(c_in_trap);
178 std::cout.flush();
179 state.reg[R_R0] = static_cast<std::uint16_t>(c_in_trap);
180 }
181 }
182 state.update_flags(R_R0);
183 break;
184 }
185 case TRAP_PUTSP: {
186 if (!state.memory.test_mode) {
187 std::uint16_t current_addr = state.reg[R_R0];
188 std::uint16_t word = state.memory.read(current_addr);
189 while (word != 0) {
190 char char1 = word & 0xFF;
191 std::cout.put(char1);
192 char char2 = (word >> 8) & 0xFF;
193 if (char2) {
194 std::cout.put(char2);
195 }
196 current_addr++;
197 word = state.memory.read(current_addr);
198 }
199 std::cout.flush();
200 }
201 break;
202 }
203 case TRAP_HALT:
204 if (!state.memory.test_mode) {
205 std::cout << "HALT" << std::endl;
206 }
207 state.running = false;
208 break;
209 default:
210 throw std::runtime_error("Unknown TRAP vector: " + std::to_string(instr & 0xFF));
211 break;
212 }
213 }

235void LC3State::load_image(const std::string &filename) {
236 std::ifstream file(filename, std::ios::binary);
237 if (!file.is_open()) {
238 throw std::runtime_error("Failed to open image file: " + filename);
239 }
240
241 std::uint16_t origin;
242 file.read(reinterpret_cast<char*>(&origin), sizeof(origin));
243 if (file.gcount() != sizeof(origin)) {
244 throw std::runtime_error("Failed to read origin from image file: " + filename);
245 }
246 origin = swap16(origin);
247
248 std::uint16_t max_words_to_read = MEMORY_MAX - origin;
249 std::vector<std::uint16_t> image_data(max_words_to_read);
250
251 file.read(reinterpret_cast<char*>(image_data.data()), max_words_to_read * sizeof(std::uint16_t));
252 size_t words_actually_read = file.gcount() / sizeof(std::uint16_t);
253
254 for (size_t i = 0; i < words_actually_read; ++i) {
255 if (origin + i < MEMORY_MAX) {
256 this->memory.write(origin + i, swap16(image_data[i]));
257 } else {
258 throw std::runtime_error("Image too large for memory at specified origin.");
259 }
260 }
261 if (words_actually_read > 0) {
262 loaded_code_segments.push_back({origin, static_cast<std::uint16_t>(words_actually_read)});
263 }
264}

281void LC3State::step() {
282 if (!this->running) return;
283
284 std::uint16_t current_pc = this->reg[R_PC];
285 this->reg[R_PC]++;
286
287 std::uint16_t instruction = this->memory.read(current_pc);
288 std::uint16_t opcode_val = instruction >> 12;
289
290 if (opcode_val >= op_table.size() || !op_table[opcode_val]) {
291 throw std::runtime_error("Illegal or unsupported opcode: " + std::to_string(opcode_val) +
292 " at PC: " + std::to_string(current_pc));
293 }
294
295 op_table[opcode_val](*this, instruction);
296}

308std::string LC3State::disassemble(std::uint16_t address) {
309 std::uint16_t instr = memory.read(address);
310 std::uint16_t opcode = instr >> 12;
311 std::ostringstream oss;
312
313 oss << "0x" << std::nouppercase << std::hex << std::setfill('0') << std::setw(4) << address << ": ";
314
315 switch (opcode) {
316 case OP_ADD: {
317 std::uint16_t dr = (instr >> 9) & 0x7;
318 std::uint16_t sr1 = (instr >> 6) & 0x7;
319 oss << "ADD R" << dr << ", R" << sr1 << ", ";
320 if ((instr >> 5) & 0x1) {
321 std::int16_t imm5 = static_cast<std::int16_t>(sign_extend(instr & 0x1F, 5));
322 oss << "#" << std::dec << imm5;
323 } else {
324 std::uint16_t sr2 = instr & 0x7;
325 oss << "R" << sr2;
326 }
327 break;
328 }
329 case OP_AND: {
330 std::uint16_t dr = (instr >> 9) & 0x7;
331 std::uint16_t sr1 = (instr >> 6) & 0x7;
332 oss << "AND R" << dr << ", R" << sr1 << ", ";
333 if ((instr >> 5) & 0x1) {
334 std::int16_t imm5 = static_cast<std::int16_t>(sign_extend(instr & 0x1F, 5));
335 oss << "#" << std::dec << imm5;
336 } else {
337 std::uint16_t sr2 = instr & 0x7;
338 oss << "R" << sr2;
339 }
340 break;
341 }
342 case OP_NOT: {
343 std::uint16_t dr = (instr >> 9) & 0x7;
344 std::uint16_t sr = (instr >> 6) & 0x7;
345 oss << "NOT R" << dr << ", R" << sr;
346 break;
347 }
348 case OP_BR: {
349 std::uint16_t n = (instr >> 11) & 0x1;
350 std::uint16_t z = (instr >> 10) & 0x1;
351 std::uint16_t p = (instr >> 9) & 0x1;
352 std::int16_t pc_offset9 = static_cast<std::int16_t>(sign_extend(instr & 0x1FF, 9));
353 oss << "BR" << (n ? "n" : "") << (z ? "z" : "") << (p ? "p" : "") << " 0x" << std::nouppercase << std::hex << std::setfill('0') << std::setw(4) << (address + 1 + pc_offset9);
354 break;
355 }
356 case OP_JMP: {
357 std::uint16_t base_r = (instr >> 6) & 0x7;
358 if (base_r == 7) {
359 oss << "RET";
360 } else {
361 oss << "JMP R" << base_r;
362 }
363 break;
364 }
365 case OP_JSR: {
366 oss << "JSR ";
367 if ((instr >> 11) & 1) {
368 std::int16_t pc_offset11 = static_cast<std::int16_t>(sign_extend(instr & 0x7FF, 11));
369 oss << "0x" << std::nouppercase << std::hex << std::setfill('0') << std::setw(4) << (address + 1 + pc_offset11);
370 } else {
371 std::uint16_t base_r = (instr >> 6) & 0x7;
372 oss << "R" << base_r;
373 }
374 break;
375 }
376 case OP_LD: {
377 std::uint16_t dr = (instr >> 9) & 0x7;
378 std::int16_t pc_offset9 = static_cast<std::int16_t>(sign_extend(instr & 0x1FF, 9));
379 oss << "LD R" << dr << ", 0x" << std::nouppercase << std::hex << std::setfill('0') << std::setw(4) << (address + 1 + pc_offset9);
380 break;
381 }
382 case OP_LDI: {
383 std::uint16_t dr = (instr >> 9) & 0x7;
384 std::int16_t pc_offset9 = static_cast<std::int16_t>(sign_extend(instr & 0x1FF, 9));
385 oss << "LDI R" << dr << ", 0x" << std::nouppercase << std::hex << std::setfill('0') << std::setw(4) << (address + 1 + pc_offset9);
386 break;
387 }
388 case OP_LDR: {
389 std::uint16_t dr = (instr >> 9) & 0x7;
390 std::uint16_t base_r = (instr >> 6) & 0x7;
391 std::int16_t offset6 = static_cast<std::int16_t>(sign_extend(instr & 0x3F, 6));
392 oss << "LDR R" << dr << ", R" << base_r << ", #" << std::dec << offset6;
393 break;
394 }
395 case OP_LEA: {
396 std::uint16_t dr = (instr >> 9) & 0x7;
397 std::int16_t pc_offset9 = static_cast<std::int16_t>(sign_extend(instr & 0x1FF, 9));
398 oss << "LEA R" << dr << ", 0x" << std::nouppercase << std::hex << std::setfill('0') << std::setw(4) << (address + 1 + pc_offset9);
399 break;
400 }
401 case OP_ST: {
402 std::uint16_t sr = (instr >> 9) & 0x7;
403 std::int16_t pc_offset9 = static_cast<std::int16_t>(sign_extend(instr & 0x1FF, 9));
404 oss << "ST R" << sr << ", 0x" << std::nouppercase << std::hex << std::setfill('0') << std::setw(4) << (address + 1 + pc_offset9);
405 break;
406 }
407 case OP_STI: {
408 std::uint16_t sr = (instr >> 9) & 0x7;
409 std::int16_t pc_offset9 = static_cast<std::int16_t>(sign_extend(instr & 0x1FF, 9));
410 oss << "STI R" << sr << ", 0x" << std::nouppercase << std::hex << std::setfill('0') << std::setw(4) << (address + 1 + pc_offset9);
411 break;
412 }
413 case OP_STR: {
414 std::uint16_t sr = (instr >> 9) & 0x7;
415 std::uint16_t base_r = (instr >> 6) & 0x7;
416 std::int16_t offset6 = static_cast<std::int16_t>(sign_extend(instr & 0x3F, 6));
417 oss << "STR R" << sr << ", R" << base_r << ", #" << std::dec << offset6;
418 break;
419 }
420 case OP_TRAP: {
421 oss << "TRAP x" << std::nouppercase << std::hex << std::setfill('0') << std::setw(2) << (instr & 0xFF);
422 break;
423 }
424 case OP_RES:
425 case OP_RTI:
426 default:
427 oss << "BAD OPCODE";
428 break;
429 }
430 return oss.str();
431}
